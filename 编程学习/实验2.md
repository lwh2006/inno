{
    float date1, date2;
    char op;

    scanf("%f%c%f", &date1, &op, &date2);
    switch(op)
    {
        case '+':
            printf("%.2f + %.2f = %.2f\n", date1, date2, date1 + date2);
            break;
        case '-':
            printf("%.2f - %.2f = %.2f\n", date1, date2, date1 - date2);
            break;
        case '*':
            printf("%.2f * %.2f = %.2f\n", date1, date2, date1 * date2);
            break;

            if(date2 == 0.0)
                printf("Division by zero!\n");

#include <stdio.h>

int main()
{
    int date1, date2;
    char op;
    char choice;

    do {
        printf("pls enter an expression: ");
        scanf("%d %c %d", &date1, &op, &date2);
        switch(op)
        {
            case '+':
                printf("%d + %d = %d\n", date1, date2, date1 + date2);
                break;
            case '-':
                printf("%d - %d = %d\n", date1, date2, date1 - date2);
                break;
            case '*':
                printf("%d * %d = %d\n", date1, date2, date1 * date2);
                break;
            case '/':
                if(date2 == 0)
                    printf("Division by zero!\n");
                else
                    printf("%d / %d = %d\n", date1, date2, date1 / date2);
                break;
            default:
                printf("Invalid operator!\n");
        }

        printf("continue？(y/n): ");
        scanf(" %c", &choice); // 注意前面的空格，用于跳过输入缓冲区中的换行符
    } while (choice == 'y' || choice == 'Y');

    return 0;
}
