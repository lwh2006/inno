#include <stdlib.h>
#include <stdio.h>
int main(void)
{
    int a;
    int b;
    int num = 0;
    char choice;
    unsigned int seed;

    do
    {
        printf("pls enter seed:");
        scanf("%u", &seed);
        srand(seed);
        a=rand()% 100 + 1;
        printf("A new random number has been generated! \n");
        do {
            printf("pls guess a number:");
            scanf("%d", &b);
            num++;
            if (b > a)
                printf("Too big!\n");

            else if (b < a)
                printf("Too small!\n");

            else
                printf("right!\n");
            if (num > 10) {
                printf("you lose!\n");
                break;
            }
        } while (b != a);
        if (b==a)
        {
            printf("ur guess number of times is %d", num);
        }
        printf("continues to guess another number? (y/n): ");
        scanf(" %c", &choice);
    }while (choice == 'y' || choice == 'Y');
    return 0;
}
